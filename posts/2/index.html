    def multi_processes_delete(self, delete_array, delete_es=False, delete_type='zeek_file'):
        logger.info('--start multi delete--')
        use_process_count = len(delete_array)
        if use_process_count > 0:
            pool = multiprocessing.Pool(processes=use_process_count)
            for i in range(use_process_count):
                per_delete_dirs = delete_array[i]
                pool.apply_async(
                    single_process_delete, (
                        per_delete_dirs,
                        self.request_dict,
                        delete_es,
                        delete_type
                    )
                )
            pool.close()
            pool.join()
        return
